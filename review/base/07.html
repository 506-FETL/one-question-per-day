  use(func) {
    if (func.length === 2)

  start(req) {
    let idx = 0
    let errIdx = 0
    const that = this

    function next(nextErr) {
      const args = [req, next]
      let func = null

      if (nextErr) {
        func = that.errHandlers[errIdx++]

        func = that.cbHandlers[idx++]

      try {
        func && Promise.resolve(func(...args)).catch(error => next(error))
      }
      catch (error) {
        next(error)
      }
    }

    next()
  }
}
